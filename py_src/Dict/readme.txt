match_result.pkl:
  2d list, length n x n
  match_result[p1][p2] = a float, show how p2's sequence performs on p1,
                         (comparing with O3)

features.pkl:
  1d list, length n
  features[p] = a 165 long vector, program's feature

benchmark_names.pkl:
  1d list, length n
  benchmark_names[p] = string, program's name
